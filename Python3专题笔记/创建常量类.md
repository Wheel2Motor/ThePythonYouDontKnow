创建常量类
================================================================================

__测试版本__：Python3.10.8

--------------------------------------------------------------------------------

Python中并没有常量的概念，比较优雅的方式就是实用元类来控制。

类是元类的实例，通过在元类中重载 __写__ 数据的方法，拦截数据的二次覆盖实现。

```python
import os

class ConstMeta(type):

    class ConstError(Exception):...

    class ConstProxyError(Exception):...

    class NotInstantiableError(Exception):...

    MAGIC_METHODS = ("_init",)

    def __new__(cls, *args, **kwargs):
        return super().__new__(cls, *args, **kwargs)

    def __init__(self, *args, **kwargs):
        vs = vars(self)
        for method in self.__class__.MAGIC_METHODS:
            if method in vs:
                getattr(self, method)()

    def __setattr__(self, key ,value):
        if key in self.__dict__:
            raise self.__class__.ConstError("Const value can't be overridden")
        super().__setattr__(key, value)

    def __getitem__(self, key):
        return vars(self)[key]

    def __call__(self, *args, **kwargs):
        raise self.__class__.NotInstantiableError("Const type can't be instantiated")

    def __repr__(self):
        prefix = "< {0} >".format(
            " ConstType: {0} "
            .format(self.__name__)
            .center(os.get_terminal_size()[0] - 4, "-")
            )
        return prefix + "\n".join(list(map(
            lambda item: str(item),
            filter(lambda item: (not item[0].startswith("__")) and
                                (not item[0] in self.__class__.MAGIC_METHODS),
                   vars(self).items()))))


if __name__ == "__main__":
    from collections import namedtuple
    class Globals(metaclass=ConstMeta):

        PLATFORM_3DSMAX = "3dsmax"
        PLATFORM_MAYA   = "maya"

        @classmethod
        def _init(cls):
            cls.PLATFORM = cls.PLATFORM_3DSMAX
            cls.VERSION = namedtuple("Version",
                 ["major", "minor", "micro", "tag", "extra"]
                 )(0, 0, 1, "release", "")

    print(Globals)
    print(Globals.VERSION)
    print(Globals["VERSION"])
```
